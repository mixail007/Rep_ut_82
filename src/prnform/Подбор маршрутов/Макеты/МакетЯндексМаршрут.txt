<script type="text/javascript">
var myArr=[ТОЧКИМАРШРУТА];
ymaps.ready(init);

function init() {
    var myMap = new ymaps.Map("map", {
            center: [57.626569, 39.893822],
            zoom: 7
        });
    myMap.controls

                // Кнопка изменения масштаба

                .add('zoomControl')

                // Список типов карты

                .add('typeSelector')

                // Стандартный набор кнопок

                .add('mapTools');
    ymaps.route(myArr).then(function (route) {
        myMap.geoObjects.add(route);
        var points = route.getWayPoints(),
        lastPoint = points.getLength() - 1;
        points.options.set('preset', 'redStretchyIcon');
        //points.options.set('preset', 'twirl#redStretchyIcon');

        //for (var i=0;i<myArr.length;i++){ 
		//points.get(i).properties.set('iconContent', myArr[i]);
		//};        
        var centerAndZoom = ymaps.util.bounds.getCenterAndZoom(
            [points.get(0).geometry.getCoordinates(),points.get(lastPoint).geometry.getCoordinates()],
            [1000,660]
        );
		var centerAndZoom2 = ymaps.util.bounds.getCenterAndZoom(
            [points.get(lastPoint).geometry.getCoordinates(),points.get(0).geometry.getCoordinates()],
            [1000,660]
        );

		if (centerAndZoom.zoom>centerAndZoom2.zoom)
			{myMap.setCenter(centerAndZoom.center, centerAndZoom.zoom);}
		else
			{myMap.setCenter(centerAndZoom2.center, centerAndZoom2.zoom);}

    }, function (error) {
        alert('Возникла ошибка: ' + error.message);
    });
}
</script>
